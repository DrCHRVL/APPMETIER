// utils/dataSync/DataMergeService.ts

import { Enquete } from '@/types/interfaces';
import { ResultatAudience } from '@/types/audienceTypes';
import { SyncData, SyncConflict, ConflictType } from '@/types/dataSyncTypes';

/**
 * Service de fusion et détection de conflits pour la synchronisation des données
 */
export class DataMergeService {
  
  /**
   * Détecte les conflits entre les données locales et serveur
   */
  static detectConflicts(localData: SyncData, serverData: SyncData): SyncConflict[] {
    const conflicts: SyncConflict[] = [];

    // 1. Conflits sur les enquêtes
    conflicts.push(...this.detectEnqueteConflicts(localData.enquetes || [], serverData.enquetes || []));

    // 2. Conflits sur les résultats d'audience
    conflicts.push(...this.detectAudienceConflicts(localData.audienceResultats || {}, serverData.audienceResultats || {}));

    // 3. Conflits sur les tags (plus simple - si différents)
    if (JSON.stringify(localData.customTags) !== JSON.stringify(serverData.customTags)) {
      conflicts.push({
        type: 'tags_modified',
        details: ['Les tags personnalisés ont été modifiés des deux côtés'],
        localData: localData.customTags,
        serverData: serverData.customTags
      });
    }

    // 4. Conflits sur les règles d'alertes
    if (JSON.stringify(localData.alertRules) !== JSON.stringify(serverData.alertRules)) {
      conflicts.push({
        type: 'rules_modified',
        details: ['Les règles d\'alertes ont été modifiées des deux côtés'],
        localData: localData.alertRules,
        serverData: serverData.alertRules
      });
    }

    return conflicts;
  }

  /**
   * Détecte les conflits sur les enquêtes
   */
  private static detectEnqueteConflicts(
    localEnquetes: Enquete[],
    serverEnquetes: Enquete[]
  ): SyncConflict[] {
    const conflicts: SyncConflict[] = [];
    const localMap = new Map((localEnquetes || []).map(e => [e.id, e]));
    const serverMap = new Map((serverEnquetes || []).map(e => [e.id, e]));

    // Vérifier chaque enquête serveur
    for (const [id, serverEnquete] of serverMap) {
      const localEnquete = localMap.get(id);

      if (!localEnquete) {
        // Nouvelle enquête sur le serveur uniquement
        continue; // Pas de conflit, on peut l'ajouter
      }

      // Comparer les dates de mise à jour
      const localDate = new Date(localEnquete.dateMiseAJour).getTime();
      const serverDate = new Date(serverEnquete.dateMiseAJour).getTime();

      // Si les deux ont été modifiés (différence > 1 seconde)
      if (Math.abs(localDate - serverDate) > 1000) {
        const details = this.compareEnquetes(localEnquete, serverEnquete);
        
        if (details.length > 0) {
          conflicts.push({
            type: 'enquete_modified',
            enqueteId: id,
            enqueteNumero: serverEnquete.numero,
            details,
            localData: localEnquete,
            serverData: serverEnquete,
            localTimestamp: localEnquete.dateMiseAJour,
            serverTimestamp: serverEnquete.dateMiseAJour
          });
        }
      }
    }

    // Vérifier les enquêtes supprimées côté serveur
    for (const [id, localEnquete] of localMap) {
      if (!serverMap.has(id)) {
        // L'enquête existe localement mais pas sur le serveur
        // Possible suppression côté serveur ou nouvelle enquête locale
        const isRecent = Date.now() - new Date(localEnquete.dateCreation).getTime() < 24 * 60 * 60 * 1000;
        
        if (!isRecent) {
          conflicts.push({
            type: 'enquete_deleted',
            enqueteId: id,
            enqueteNumero: localEnquete.numero,
            details: ['Enquête supprimée sur le serveur mais présente localement'],
            localData: localEnquete
          });
        }
      }
    }

    return conflicts;
  }

  /**
   * Compare deux enquêtes et retourne la liste des différences
   */
  private static compareEnquetes(local: Enquete, server: Enquete): string[] {
    const differences: string[] = [];

    // Comparer les champs principaux
    if (local.numero !== server.numero) {
      differences.push(`Numéro modifié: "${local.numero}" vs "${server.numero}"`);
    }

    if (local.statut !== server.statut) {
      differences.push(`Statut modifié: "${local.statut}" vs "${server.statut}"`);
    }

    if (local.description !== server.description) {
      differences.push('Description modifiée');
    }

    // Comparer les comptes-rendus
    if (local.comptesRendus.length !== server.comptesRendus.length) {
      differences.push(`Comptes-rendus: ${local.comptesRendus.length} local vs ${server.comptesRendus.length} serveur`);
    }

    // Comparer les mis en cause
    if (local.misEnCause.length !== server.misEnCause.length) {
      differences.push(`Mis en cause: ${local.misEnCause.length} local vs ${server.misEnCause.length} serveur`);
    }

    // Comparer les actes
    const localActesCount = (local.actes?.length || 0) + (local.ecoutes?.length || 0) + (local.geolocalisations?.length || 0);
    const serverActesCount = (server.actes?.length || 0) + (server.ecoutes?.length || 0) + (server.geolocalisations?.length || 0);
    
    if (localActesCount !== serverActesCount) {
      differences.push(`Actes: ${localActesCount} local vs ${serverActesCount} serveur`);
    }

    // Comparer les documents
    if ((local.documents?.length || 0) !== (server.documents?.length || 0)) {
      differences.push(`Documents: ${local.documents?.length || 0} local vs ${server.documents?.length || 0} serveur`);
    }

    // Comparer les tags
    if (JSON.stringify(local.tags) !== JSON.stringify(server.tags)) {
      differences.push('Tags modifiés');
    }

    return differences;
  }

  /**
   * Détecte les conflits sur les résultats d'audience
   */
  private static detectAudienceConflicts(
    localResults: Record<string, ResultatAudience>,
    serverResults: Record<string, ResultatAudience>
  ): SyncConflict[] {
    const conflicts: SyncConflict[] = [];

    for (const [enqueteId, serverResult] of Object.entries(serverResults)) {
      const localResult = localResults[enqueteId];

      if (!localResult) {
        continue; // Nouveau résultat serveur, pas de conflit
      }

      // Comparer les résultats
      if (JSON.stringify(localResult) !== JSON.stringify(serverResult)) {
        conflicts.push({
          type: 'audience_modified',
          enqueteId: parseInt(enqueteId),
          details: ['Résultat d\'audience modifié des deux côtés'],
          localData: localResult,
          serverData: serverResult
        });
      }
    }

    return conflicts;
  }

  /**
   * Fusionne les données sans conflit (prend le plus récent)
   */
  static mergeWithoutConflicts(localData: SyncData, serverData: SyncData): SyncData {
    const merged: SyncData = {
      enquetes: this.mergeEnquetes(localData.enquetes || [], serverData.enquetes || []),
      audienceResultats: this.mergeAudienceResults(localData.audienceResultats, serverData.audienceResultats),
      customTags: { ...serverData.customTags, ...localData.customTags }, // Local prioritaire pour les tags
      alertRules: this.mergeAlertRules(localData.alertRules, serverData.alertRules),
      version: Math.max(localData.version, serverData.version) + 1
    };

    return merged;
  }

  /**
   * Fusionne les enquêtes (le plus récent gagne)
   */
  private static mergeEnquetes(localEnquetes: Enquete[], serverEnquetes: Enquete[]): Enquete[] {
    const merged = new Map<number, Enquete>();

    // Ajouter toutes les enquêtes locales
    localEnquetes.forEach(enquete => {
      merged.set(enquete.id, enquete);
    });

    // Fusionner avec les enquêtes serveur
    serverEnquetes.forEach(serverEnquete => {
      const localEnquete = merged.get(serverEnquete.id);

      if (!localEnquete) {
        // Nouvelle enquête du serveur
        merged.set(serverEnquete.id, serverEnquete);
      } else {
        // Comparer les dates de mise à jour
        const localDate = new Date(localEnquete.dateMiseAJour).getTime();
        const serverDate = new Date(serverEnquete.dateMiseAJour).getTime();

        if (serverDate > localDate) {
          // Version serveur plus récente
          merged.set(serverEnquete.id, serverEnquete);
        }
        // Sinon garder la version locale
      }
    });

    return Array.from(merged.values());
  }

  /**
   * Fusionne les résultats d'audience
   */
  private static mergeAudienceResults(
    local: Record<string, ResultatAudience>,
    server: Record<string, ResultatAudience>
  ): Record<string, ResultatAudience> {
    return { ...server, ...local }; // Local prioritaire
  }

  /**
   * Fusionne les règles d'alertes
   */
  private static mergeAlertRules(local: any[], server: any[]): any[] {
    const merged = new Map();

    // Ajouter les règles serveur
    server.forEach(rule => merged.set(rule.id, rule));

    // Ajouter/écraser avec les règles locales
    local.forEach(rule => merged.set(rule.id, rule));

    return Array.from(merged.values());
  }

  /**
   * Applique la résolution "keep_local" - garde les données locales
   */
  static resolveKeepLocal(localData: SyncData, serverData: SyncData): SyncData {
    // On garde toutes les données locales mais on ajoute les nouvelles enquêtes du serveur
    const serverEnqueteIds = new Set((serverData.enquetes || []).map(e => e.id));
    const localEnqueteIds = new Set((localData.enquetes || []).map(e => e.id));

    // Trouver les nouvelles enquêtes serveur qui n'existent pas localement
    const newServerEnquetes = (serverData.enquetes || []).filter(e => !localEnqueteIds.has(e.id));

    return {
      enquetes: [...localData.enquetes, ...newServerEnquetes],
      audienceResultats: { ...serverData.audienceResultats, ...localData.audienceResultats },
      customTags: localData.customTags,
      alertRules: localData.alertRules,
      version: localData.version + 1
    };
  }

  /**
   * Applique la résolution "keep_server" - garde les données serveur
   */
  static resolveKeepServer(localData: SyncData, serverData: SyncData): SyncData {
    // On garde les données serveur mais on ajoute les nouvelles enquêtes locales
    const localEnqueteIds = new Set((localData.enquetes || []).map(e => e.id));
    const serverEnqueteIds = new Set((serverData.enquetes || []).map(e => e.id));

    const newLocalEnquetes = (localData.enquetes || []).filter(e => !serverEnqueteIds.has(e.id));

    return {
      enquetes: [...serverData.enquetes, ...newLocalEnquetes],
      audienceResultats: { ...localData.audienceResultats, ...serverData.audienceResultats },
      customTags: serverData.customTags,
      alertRules: serverData.alertRules,
      version: serverData.version + 1
    };
  }

  /**
   * Formate les conflits en message lisible
   */
  static formatConflictsMessage(conflicts: SyncConflict[]): string {
    const lines: string[] = [
      '⚠️ CONFLITS DÉTECTÉS',
      '',
      `${conflicts.length} conflit(s) détecté(s) entre vos données locales et le serveur commun :`,
      ''
    ];

    conflicts.forEach((conflict, index) => {
      lines.push(`${index + 1}. ${this.getConflictTypeLabel(conflict.type)}`);
      
      if (conflict.enqueteNumero) {
        lines.push(`   Enquête: ${conflict.enqueteNumero}`);
      }
      
      conflict.details.forEach(detail => {
        lines.push(`   • ${detail}`);
      });
      
      lines.push('');
    });

    lines.push('Voulez-vous ÉCRASER les données du serveur avec vos modifications locales ?');
    lines.push('');
    lines.push('⚠️ Attention: Cette action est irréversible !');
    lines.push('');
    lines.push('Cliquez OK pour écraser le serveur, Annuler pour intégrer les données serveur localement.');

    return lines.join('\n');
  }

  /**
   * Retourne le libellé d'un type de conflit
   */
  private static getConflictTypeLabel(type: ConflictType): string {
    const labels: Record<ConflictType, string> = {
      'enquete_modified': 'Enquête modifiée des deux côtés',
      'enquete_deleted': 'Enquête supprimée sur le serveur',
      'enquete_new': 'Nouvelle enquête avec ID identique',
      'audience_modified': 'Résultat d\'audience modifié',
      'tags_modified': 'Tags personnalisés modifiés',
      'rules_modified': 'Règles d\'alertes modifiées'
    };

    return labels[type] || 'Conflit inconnu';
  }
}
