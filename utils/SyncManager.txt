// utils/SyncManager.ts
import { UserIdentification } from './UserIdentification';
import { ElectronBridge } from './electronBridge';
import { SyncData, SyncResult, SyncStatus, ChangeNotification, SyncMetadata } from '../types/syncTypes';
import { Enquete } from '../types/interfaces';

class SyncManagerService {
  private static instance: SyncManagerService;
  private currentUser: string = 'Utilisateur Inconnu';
  private isInitialized = false;
  private syncInterval: NodeJS.Timeout | null = null;
  private statusListeners: Array<(status: SyncStatus) => void> = [];
  private changeListeners: Array<(notification: ChangeNotification) => void> = [];
  private isCurrentlySync = false;
  private lastSyncAttempt: string | null = null;
  private lastSuccessfulSync: string | null = null;
  private isOnline = false;
  
  // 🆕 LIMITATION DES TENTATIVES
  private consecutiveFailures = 0;
  private readonly MAX_CONSECUTIVE_FAILURES = 3;
  private readonly FAILURE_BACKOFF_TIME = 5 * 60 * 1000; // 5 minutes
  private failureBackoffUntil: Date | null = null;
  private isInFailureBackoff = false;

  private constructor() {}

  public static getInstance(): SyncManagerService {
    if (!SyncManagerService.instance) {
      SyncManagerService.instance = new SyncManagerService();
    }
    return SyncManagerService.instance;
  }

  // 🆕 VÉRIFICATION BACKOFF
  private isInBackoffPeriod(): boolean {
    if (!this.failureBackoffUntil) return false;
    const now = new Date();
    
    if (now < this.failureBackoffUntil) {
      return true;
    } else {
      // Reset backoff
      this.failureBackoffUntil = null;
      this.isInFailureBackoff = false;
      this.consecutiveFailures = 0;
      return false;
    }
  }

  // 🆕 GESTION DES ÉCHECS CONSÉCUTIFS
  private handleSyncFailure() {
    this.consecutiveFailures++;
    console.warn(`🚨 Échec sync ${this.consecutiveFailures}/${this.MAX_CONSECUTIVE_FAILURES}`);
    
    if (this.consecutiveFailures >= this.MAX_CONSECUTIVE_FAILURES) {
      this.failureBackoffUntil = new Date(Date.now() + this.FAILURE_BACKOFF_TIME);
      this.isInFailureBackoff = true;
      console.warn(`⏸️ Mode backoff activé jusqu'à ${this.failureBackoffUntil.toLocaleTimeString()}`);
    }
  }

  // 🆕 RESET DES ÉCHECS LORS D'UN SUCCÈS
  private handleSyncSuccess() {
    if (this.consecutiveFailures > 0) {
      console.log(`✅ Sync réussie après ${this.consecutiveFailures} échec(s)`);
    }
    this.consecutiveFailures = 0;
    this.failureBackoffUntil = null;
    this.isInFailureBackoff = false;
  }

  public async startSyncService(): Promise<void> {
    if (this.isInitialized) {
      console.log('🔄 Service de synchronisation déjà initialisé');
      return;
    }

    try {
      console.log('🚀 Démarrage du service de synchronisation...');
      
      // 🔧 CORRECTION : Identifier l'utilisateur ASYNC
      const userInfo = await UserIdentification.getCurrentUser();
      this.currentUser = userInfo.displayName;
      console.log('🔐 Utilisateur identifié:', this.currentUser, `(${userInfo.computerName})`);
      
      // 🆕 VÉRIFICATION ACCÈS SERVEUR AVEC PROTECTION
      await this.checkServerAccess();
      
      // Synchronisation initiale seulement si pas en backoff
      if (!this.isInBackoffPeriod()) {
        console.log('🔄 Synchronisation initial en cours...');
        const initialSyncResult = await this.performSync('initial');
        
        if (initialSyncResult.success) {
          this.handleSyncSuccess();
        } else {
          this.handleSyncFailure();
        }
      } else {
        console.log('⏸️ Synchronisation initiale ignorée (mode backoff)');
      }
      
      // Démarrer le cycle périodique avec protection
      this.startPeriodicSync();
      
      this.isInitialized = true;
      this.notifyStatusChange();
      
      console.log('✅ Service de synchronisation démarré');
    } catch (error) {
      console.error('❌ Erreur initialisation SyncManager:', error);
      this.handleSyncFailure();
      this.notifyStatusChange();
    }
  }

  public stopSyncService(): void {
    if (this.syncInterval) {
      clearInterval(this.syncInterval);
      this.syncInterval = null;
    }
    
    this.isInitialized = false;
    this.isCurrentlySync = false;
    
    console.log('🛑 Arrêt du service de synchronisation...');
    this.notifyStatusChange();
  }

  private startPeriodicSync(): void {
    // Nettoyer l'ancien intervalle
    if (this.syncInterval) {
      clearInterval(this.syncInterval);
    }
    
    // 🆕 INTERVALLE PLUS LONG + PROTECTION BACKOFF
    this.syncInterval = setInterval(async () => {
      // Vérifier si on est en backoff
      if (this.isInBackoffPeriod()) {
        console.log('⏸️ Sync périodique ignorée (mode backoff)');
        return;
      }

      // Éviter la concurrence
      if (!this.isCurrentlySync && this.isOnline) {
        try {
          const result = await this.performSync('periodic');
          if (result.success) {
            this.handleSyncSuccess();
          } else {
            this.handleSyncFailure();
          }
        } catch (error) {
          console.error('❌ Erreur sync périodique:', error);
          this.handleSyncFailure();
        }
      }
    }, 5 * 60 * 1000); // 🆕 5 MINUTES AU LIEU DE 30 SECONDES
  }

  private async checkServerAccess(): Promise<void> {
    try {
      // 🆕 PROTECTION CONTRE LES VÉRIFICATIONS EXCESSIVES
      if (!ElectronBridge.isAvailable()) {
        this.isOnline = false;
        console.log('🚫 APIs Electron non disponibles');
        return;
      }

      this.isOnline = await window.electronAPI.checkCommonServerAccess();
      
      if (this.isOnline) {
        console.log('🌐 Serveur commun accessible');
      } else {
        console.log('🚫 Serveur commun inaccessible');
      }
    } catch (error) {
      console.error('❌ Erreur vérification serveur:', error);
      this.isOnline = false;
    }
    
    this.notifyStatusChange();
  }

  public async triggerImmediateSync(): Promise<SyncResult> {
    // 🆕 PROTECTION BACKOFF POUR SYNC IMMÉDIATE
    if (this.isInBackoffPeriod()) {
      console.log('⏸️ Sync immédiate refusée (mode backoff)');
      return {
        success: false,
        timestamp: new Date().toISOString(),
        changesDetected: false,
        errors: ['Service temporairement indisponible (trop d\'échecs consécutifs)']
      };
    }

    if (this.isCurrentlySync) {
      console.log('⚠️ Sync déjà en cours, ignorée');
      return {
        success: false,
        timestamp: new Date().toISOString(),
        changesDetected: false,
        errors: ['Synchronisation déjà en cours']
      };
    }

    // Délai avant sync pour permettre le groupement d'actions
    await new Promise(resolve => setTimeout(resolve, 2000));
    
    try {
      const result = await this.performSync('immediate');
      
      if (result.success) {
        this.handleSyncSuccess();
      } else {
        this.handleSyncFailure();
      }
      
      return result;
    } catch (error) {
      console.error('❌ Erreur sync immédiate:', error);
      this.handleSyncFailure();
      
      return {
        success: false,
        timestamp: new Date().toISOString(),
        changesDetected: false,
        errors: [error instanceof Error ? error.message : 'Erreur inconnue']
      };
    }
  }

  private async performSync(type: 'initial' | 'periodic' | 'immediate'): Promise<SyncResult> {
    this.isCurrentlySync = true;
    this.lastSyncAttempt = new Date().toISOString();
    this.notifyStatusChange();

    try {
      // Vérifier la disponibilité
      if (!this.isOnline) {
        await this.checkServerAccess();
        if (!this.isOnline) {
          throw new Error('Serveur commun inaccessible');
        }
      }

      // Récupérer les données locales
      const localData = await this.getLocalData();
      
      // Récupérer les données du serveur
      const serverResponse = await this.getServerData();
      
      let changesDetected = false;
      
      if (serverResponse) {
        // Fusionner et synchroniser
        const mergeResult = await this.mergeAndSync(localData, serverResponse);
        changesDetected = mergeResult.changesDetected;
      } else {
        // Premier sync - envoyer nos données
        await this.sendDataToServer(localData);
      }

      this.lastSuccessfulSync = new Date().toISOString();
      
      console.log(`✅ Synchronisation ${type} réussie`);
      
      return {
        success: true,
        timestamp: this.lastSuccessfulSync,
        changesDetected,
        errors: []
      };

    } catch (error) {
      console.error(`❌ Erreur sync ${type}:`, error);
      
      return {
        success: false,
        timestamp: new Date().toISOString(),
        changesDetected: false,
        errors: [error instanceof Error ? error.message : 'Erreur inconnue']
      };
    } finally {
      this.isCurrentlySync = false;
      this.notifyStatusChange();
    }
  }

  private async getLocalData(): Promise<SyncData> {
    try {
      const enquetes = await ElectronBridge.getData('enquetes', []);
      const customTags = await ElectronBridge.getData('tags', {});
      
      return {
        enquetes: Array.isArray(enquetes) ? enquetes : [],
        customTags: customTags || {},
        lastSyncTimestamp: new Date().toISOString()
      };
    } catch (error) {
      console.error('❌ Erreur chargement données locales:', error);
      throw error;
    }
  }

  private async getServerData(): Promise<{ data: SyncData; metadata: SyncMetadata } | null> {
    try {
      return await window.electronAPI.syncFromCommonServer();
    } catch (error) {
      console.error('❌ Erreur lecture serveur:', error);
      throw error;
    }
  }

  private async sendDataToServer(data: SyncData): Promise<void> {
    try {
      // 🔧 CORRECTION : Récupérer userInfo async
      const userInfo = await UserIdentification.getCurrentUser();
      
      const metadata: SyncMetadata = {
        lastModified: new Date().toISOString(),
        modifiedBy: this.currentUser,
        version: 1,
        computerName: userInfo.computerName
      };

      const success = await window.electronAPI.syncToCommonServer(data, metadata);
      
      if (!success) {
        throw new Error('Échec envoi vers serveur commun');
      }
      
      console.log('📤 Données envoyées vers serveur commun');
    } catch (error) {
      console.error('❌ Erreur envoi serveur:', error);
      throw error;
    }
  }

  private async mergeAndSync(localData: SyncData, serverResponse: { data: SyncData; metadata: SyncMetadata }): Promise<{ changesDetected: boolean }> {
    try {
      const { data: serverData } = serverResponse;
      
      // Fusionner les enquêtes (dernier qui modifie gagne)
      const mergedEnquetes = this.mergeEnquetes(localData.enquetes, serverData.enquetes);
      
      // Détecter les changements par d'autres utilisateurs
      const changesDetected = this.generateChangeNotifications(localData.enquetes, serverData.enquetes);
      
      // Sauvegarder localement les données fusionnées
      await ElectronBridge.setData('enquetes', mergedEnquetes);
      
      // Renvoyer vers le serveur si on a des modifications locales
      const hasLocalChanges = localData.lastSyncTimestamp > serverResponse.metadata.lastModified;
      
      if (hasLocalChanges) {
        await this.sendDataToServer({
          ...localData,
          enquetes: mergedEnquetes
        });
      }
      
      return { changesDetected };
    } catch (error) {
      console.error('❌ Erreur fusion données:', error);
      throw error;
    }
  }

  private mergeEnquetes(localEnquetes: Enquete[], serverEnquetes: Enquete[]): Enquete[] {
    const merged = new Map<number, Enquete>();
    
    // Ajouter toutes les enquêtes locales
    localEnquetes.forEach(enquete => {
      merged.set(enquete.id, enquete);
    });
    
    // Fusionner avec les enquêtes du serveur
    serverEnquetes.forEach(serverEnquete => {
      const localEnquete = merged.get(serverEnquete.id);
      
      if (!localEnquete) {
        // Nouvelle enquête du serveur
        merged.set(serverEnquete.id, serverEnquete);
      } else {
        // Comparer les dates de mise à jour
        const localDate = new Date(localEnquete.dateMiseAJour);
        const serverDate = new Date(serverEnquete.dateMiseAJour);
        
        if (serverDate > localDate) {
          // La version serveur est plus récente
          merged.set(serverEnquete.id, serverEnquete);
        }
        // Sinon, garder la version locale
      }
    });
    
    return Array.from(merged.values());
  }

  private generateChangeNotifications(localEnquetes: Enquete[], serverEnquetes: Enquete[]): boolean {
    let hasChanges = false;
    
    serverEnquetes.forEach(serverEnquete => {
      const localEnquete = localEnquetes.find(e => e.id === serverEnquete.id);
      
      if (!localEnquete) {
        // Nouvelle enquête
        this.notifyChange({
          id: `new-${serverEnquete.id}-${Date.now()}`,
          type: 'enquete_added',
          enqueteId: serverEnquete.id,
          enqueteNumero: serverEnquete.numero,
          timestamp: new Date().toISOString(),
          modifiedBy: 'Collaborateur',
          description: `Nouvelle enquête "${serverEnquete.numero}"`
        });
        hasChanges = true;
      } else if (new Date(serverEnquete.dateMiseAJour) > new Date(localEnquete.dateMiseAJour)) {
        // Enquête modifiée
        this.notifyChange({
          id: `update-${serverEnquete.id}-${Date.now()}`,
          type: 'enquete_updated',
          enqueteId: serverEnquete.id,
          enqueteNumero: serverEnquete.numero,
          timestamp: new Date().toISOString(),
          modifiedBy: 'Collaborateur',
          description: `Enquête "${serverEnquete.numero}" modifiée`
        });
        hasChanges = true;
      }
    });
    
    return hasChanges;
  }

  public getSyncStatus(): SyncStatus {
    return {
      isOnline: this.isOnline && !this.isInFailureBackoff,
      lastSyncAttempt: this.lastSyncAttempt,
      lastSuccessfulSync: this.lastSuccessfulSync,
      isSync: this.isCurrentlySync,
      currentUser: this.currentUser
    };
  }

  // Listeners
  public addStatusListener(listener: (status: SyncStatus) => void): void {
    this.statusListeners.push(listener);
  }

  public removeStatusListener(listener: (status: SyncStatus) => void): void {
    const index = this.statusListeners.indexOf(listener);
    if (index > -1) {
      this.statusListeners.splice(index, 1);
    }
  }

  public addChangeListener(listener: (notification: ChangeNotification) => void): void {
    this.changeListeners.push(listener);
  }

  public removeChangeListener(listener: (notification: ChangeNotification) => void): void {
    const index = this.changeListeners.indexOf(listener);
    if (index > -1) {
      this.changeListeners.splice(index, 1);
    }
  }

  private notifyStatusChange(): void {
    const status = this.getSyncStatus();
    this.statusListeners.forEach(listener => {
      try {
        listener(status);
      } catch (error) {
        console.error('Erreur listener status:', error);
      }
    });
  }

  private notifyChange(notification: ChangeNotification): void {
    this.changeListeners.forEach(listener => {
      try {
        listener(notification);
      } catch (error) {
        console.error('Erreur listener change:', error);
      }
    });
  }
}

export const SyncManager = SyncManagerService.getInstance();
