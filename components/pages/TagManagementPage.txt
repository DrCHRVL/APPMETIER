"use client"

import React, { useState, useRef, useEffect } from 'react';
import { Card, CardHeader, CardTitle, CardContent } from '../ui/card';
import { Input } from '../ui/input';
import { Button } from '../ui/button';
import { Badge } from '../ui/badge';
import { Select } from '../ui/select';
import { Edit2, Save, X, Plus, Check, Settings } from 'lucide-react';
import { TAG_CATEGORIES, TagCategory, SERVICE_FAMILIES, ServiceTag, PREDEFINED_TAGS, ServiceFamily } from '@/config/tags';
import { Dialog, DialogContent, DialogHeader, DialogTitle, DialogFooter } from '../ui/dialog';
import { Tabs, TabsContent, TabsList, TabsTrigger } from '../ui/tabs';
import { ServiceFamilyManager } from '../ServiceFamilyManager';
import { useAtom } from 'jotai';
import { customTagsAtom } from '@/atoms/tags';
import { useToast } from '@/contexts/ToastContext';
import { ElectronBridge } from '@/utils/electronBridge';
import { APP_CONFIG } from '@/config/constants';

interface TagManagementPageProps {
  onUpdateTags: (category: TagCategory, tags: string[]) => void;
}

interface EditingTag {
  id: string;
  category: TagCategory;
  originalValue: string;
  newValue: string;
}

interface EditingServiceDetails {
  tagId: string;
  family: string;
  order: number;
}

export const TagManagementPage = ({ onUpdateTags }: TagManagementPageProps) => {
  const { showToast } = useToast();
  const [customTags, setCustomTags] = useAtom(customTagsAtom);
  const [editingCategory, setEditingCategory] = useState<TagCategory | null>(null);
  const [editingTag, setEditingTag] = useState<EditingTag | null>(null);
  const [editingServiceDetails, setEditingServiceDetails] = useState<EditingServiceDetails | null>(null);
  const [newTagDialog, setNewTagDialog] = useState(false);
  const [newTag, setNewTag] = useState('');
  const [newDuration, setNewDuration] = useState('');
  const [newServiceFamily, setNewServiceFamily] = useState('sr_amiens');
  const [newServiceOrder, setNewServiceOrder] = useState(1);
  const [serviceFamilies, setServiceFamilies] = useState<ServiceFamily[]>(SERVICE_FAMILIES);
  const editInputRef = useRef<HTMLInputElement>(null);

  useEffect(() => {
    if (editingTag && editInputRef.current) {
      editInputRef.current.focus();
      editInputRef.current.select();
    }
  }, [editingTag]);

  const getPlaceholder = (category: TagCategory) => {
    switch (category) {
      case 'services':
        return 'Nom du nouveau service d\'enquête';
      case 'infractions':
        return 'Nouveau type d\'infraction';
      case 'duree':
        return 'Nouvelle durée';
      default:
        return 'Nom du tag';
    }
  };

  const handleStartEdit = (category: TagCategory) => {
    setEditingCategory(category);
  };

  const handleSaveEdit = (category: TagCategory) => {
    onUpdateTags(category, customTags[category]);
    setEditingCategory(null);
    showToast('Modifications enregistrées', 'success');
  };

  const handleStartTagEdit = (tag: any, category: TagCategory) => {
    const tagValue = tag.value || tag;
    const tagId = tag.id || tag.value || tag;
    
    setEditingTag({
      id: tagId,
      category,
      originalValue: tagValue,
      newValue: tagValue
    });
  };

  const handleSaveTagEdit = async () => {
    if (!editingTag) return;

    const { id, category, originalValue, newValue } = editingTag;
    
    if (newValue.trim() === '') {
      showToast('Le nom du tag ne peut pas être vide', 'error');
      return;
    }

    if (newValue.trim() === originalValue) {
      setEditingTag(null);
      return;
    }

    const existingTags = customTags[category] || [];
    const tagExists = existingTags.some((tag: any) => {
      const tagValue = tag.value || tag;
      const tagId = tag.id || tag.value || tag;
      return tagValue.toLowerCase() === newValue.trim().toLowerCase() && tagId !== id;
    });

    if (tagExists) {
      showToast('Un tag avec ce nom existe déjà', 'error');
      return;
    }

    try {
      const updatedTags = JSON.parse(JSON.stringify(customTags));
      const tagIndex = updatedTags[category].findIndex((tag: any) => {
        const tagId = tag.id || tag.value || tag;
        return tagId === id;
      });

      if (tagIndex !== -1) {
        if (typeof updatedTags[category][tagIndex] === 'string') {
          updatedTags[category][tagIndex] = {
            id: id,
            value: newValue.trim(),
            category: category
          };
        } else {
          updatedTags[category][tagIndex].value = newValue.trim();
        }
      }

      await updateEnquetesWithNewTagValue(category, originalValue, newValue.trim());
      setCustomTags(updatedTags);
      setEditingTag(null);
      showToast(`Tag "${originalValue}" renommé en "${newValue.trim()}"`, 'success');
    } catch (error) {
      console.error('Erreur lors de la modification du tag:', error);
      showToast('Erreur lors de la modification du tag', 'error');
    }
  };

  const updateEnquetesWithNewTagValue = async (category: TagCategory, oldValue: string, newValue: string) => {
    try {
      const enquetes = await ElectronBridge.getData(APP_CONFIG.STORAGE_KEYS.ENQUETES, []);
      if (!Array.isArray(enquetes)) return;

      let updatedCount = 0;
      const updatedEnquetes = enquetes.map((enquete: any) => {
        if (!enquete.tags || !Array.isArray(enquete.tags)) return enquete;

        const updatedTags = enquete.tags.map((tag: any) => {
          const tagValue = tag.value || tag;
          const tagCategory = tag.category;

          if (tagCategory === category && tagValue === oldValue) {
            updatedCount++;
            return {
              ...tag,
              value: newValue,
              id: tag.id || `${category}-${newValue}`
            };
          }
          return tag;
        });

        return {
          ...enquete,
          tags: updatedTags,
          dateMiseAJour: new Date().toISOString()
        };
      });

      if (updatedCount > 0) {
        await ElectronBridge.setData(APP_CONFIG.STORAGE_KEYS.ENQUETES, updatedEnquetes);
      }
    } catch (error) {
      console.error('Erreur lors de la mise à jour des enquêtes:', error);
      throw error;
    }
  };

  const handleUpdateServiceFamilies = (newFamilies: ServiceFamily[]) => {
    setServiceFamilies(newFamilies);
    // Sauvegarder dans le stockage
    ElectronBridge.setData('service_families', newFamilies).catch(console.error);
  };

  const handleUpdateServices = (newServices: ServiceTag[]) => {
    const updatedTags = { ...customTags };
    updatedTags.services = newServices;
    setCustomTags(updatedTags);
  };

  const handleCancelTagEdit = () => {
    setEditingTag(null);
  };

  const handleEditKeyDown = (e: React.KeyboardEvent) => {
    if (e.key === 'Enter') {
      handleSaveTagEdit();
    } else if (e.key === 'Escape') {
      handleCancelTagEdit();
    }
  };

  const handleEditServiceDetails = (tag: any) => {
    const tagValue = tag.value || tag;
    const tagId = tag.id || tag.value || tag;
    
    const serviceTag = PREDEFINED_TAGS.services.find(s => s.value === tagValue);
    const currentFamily = tag.family || serviceTag?.family || 'sr_amiens';
    const currentOrder = tag.order || serviceTag?.order || 1;
    
    setEditingServiceDetails({
      tagId: tagId,
      family: currentFamily,
      order: currentOrder
    });
  };

  const handleSaveServiceDetails = async () => {
    if (!editingServiceDetails) return;

    try {
      const updatedTags = JSON.parse(JSON.stringify(customTags));
      const tagIndex = updatedTags.services.findIndex((tag: any) => {
        const tagId = tag.id || tag.value || tag;
        return tagId === editingServiceDetails.tagId;
      });

      if (tagIndex !== -1) {
        if (typeof updatedTags.services[tagIndex] === 'string') {
          updatedTags.services[tagIndex] = {
            id: editingServiceDetails.tagId,
            value: updatedTags.services[tagIndex],
            category: 'services',
            family: editingServiceDetails.family,
            order: editingServiceDetails.order
          };
        } else {
          updatedTags.services[tagIndex] = {
            ...updatedTags.services[tagIndex],
            family: editingServiceDetails.family,
            order: editingServiceDetails.order
          };
        }
      }

      setCustomTags(updatedTags);
      setEditingServiceDetails(null);
      showToast('Détails du service mis à jour', 'success');
    } catch (error) {
      console.error('Erreur lors de la mise à jour:', error);
      showToast('Erreur lors de la mise à jour', 'error');
    }
  };

  const handleAddTag = (category: TagCategory, customTagValue?: string) => {
    const tagToAdd = customTagValue || newTag.trim();
    if (!tagToAdd) return;

    if (category === 'duree') {
      if (!tagToAdd.startsWith('enquête de plus de') && tagToAdd !== 'enquête à venir') {
        showToast('Le tag de durée doit commencer par "enquête de plus de" ou être "enquête à venir"', 'error');
        return;
      }

      if (tagToAdd !== 'enquête à venir' && !tagToAdd.includes('jours') && !newDuration) {
        showToast('Veuillez spécifier une durée', 'error');
        return;
      }
    }

    const existingTags = customTags[category] || [];
    const tagExists = existingTags.some((tag: any) => {
      const tagValue = tag.value || tag;
      return tagValue.toLowerCase() === tagToAdd.toLowerCase();
    });

    if (tagExists) {
      showToast('Ce tag existe déjà', 'error');
      return;
    }

    let newTagObject: any = {
      id: `tag_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`,
      value: tagToAdd,
      category: category
    };

    if (category === 'services') {
      newTagObject.family = newServiceFamily;
      newTagObject.order = newServiceOrder;
    }

    const updatedTagsObject = JSON.parse(JSON.stringify(customTags));
    if (!Array.isArray(updatedTagsObject[category])) {
      updatedTagsObject[category] = [];
    }
    updatedTagsObject[category].push(newTagObject);

    setCustomTags(updatedTagsObject);
    setNewTag('');
    setNewDuration('');
    setNewServiceFamily('sr_amiens');
    setNewServiceOrder(1);
    setNewTagDialog(false);
    showToast('Tag ajouté avec succès', 'success');
  };

  const handleRemoveTag = async (category: TagCategory, tagId: string) => {
    try {
      const tagToRemove = customTags[category]?.find((tag: any) => {
        const currentTagId = tag.id || tag.value || tag;
        return currentTagId === tagId;
      });
      
      if (!tagToRemove) {
        showToast('Tag non trouvé', 'error');
        return;
      }

      const tagValue = tagToRemove.value || tagToRemove;
      const enquetes = await ElectronBridge.getData(APP_CONFIG.STORAGE_KEYS.ENQUETES, []);
      const usageCount = enquetes.filter((enquete: any) => 
        enquete.tags?.some((tag: any) => {
          const currentTagValue = tag.value || tag;
          return currentTagValue === tagValue;
        })
      ).length;

      if (usageCount > 0) {
        const confirmDelete = window.confirm(
          `Ce tag est utilisé dans ${usageCount} enquête(s). Êtes-vous sûr de vouloir le supprimer ?`
        );
        if (!confirmDelete) return;

        const updatedEnquetes = enquetes.map((enquete: any) => ({
          ...enquete,
          tags: enquete.tags?.filter((tag: any) => {
            const currentTagValue = tag.value || tag;
            return currentTagValue !== tagValue;
          }) || [],
          dateMiseAJour: new Date().toISOString()
        }));

        await ElectronBridge.setData(APP_CONFIG.STORAGE_KEYS.ENQUETES, updatedEnquetes);
      }

      const updatedTags = JSON.parse(JSON.stringify(customTags));
      if (!updatedTags[category] || !Array.isArray(updatedTags[category])) {
        updatedTags[category] = [];
      } else {
        updatedTags[category] = updatedTags[category].filter((tag: any) => {
          const currentTagId = tag.id || tag.value || tag;
          return currentTagId !== tagId;
        });
      }
      
      setCustomTags(updatedTags);
      showToast(usageCount > 0 ? 
        `Tag supprimé de ${usageCount} enquête(s)` : 
        'Tag supprimé avec succès', 
        'success'
      );
    } catch (error) {
      console.error('Erreur lors de la suppression du tag:', error);
      showToast('Erreur lors de la suppression du tag', 'error');
    }
  };

  const renderServiceTag = (tag: any, category: TagCategory) => {
    const tagValue = tag.value || tag;
    const tagId = tag.id || tag.value || tag;
    const isCurrentlyEditing = editingTag?.id === tagId;
    const isEditingDetails = editingServiceDetails?.tagId === tagId;
    
    const serviceTag = PREDEFINED_TAGS.services.find(s => s.value === tagValue);
    const family = tag.family || serviceTag?.family || 'sr_amiens';
    const order = tag.order || serviceTag?.order || 1;
    const familyName = serviceFamilies.find(f => f.id === family)?.name || family;

    if (isCurrentlyEditing) {
      return (
        <div key={tagId} className="flex items-center">
          <div className="flex items-center gap-1 bg-white border rounded px-2 py-1">
            <Input
              ref={editInputRef}
              value={editingTag!.newValue}
              onChange={(e) => setEditingTag({
                ...editingTag!,
                newValue: e.target.value
              })}
              onKeyDown={handleEditKeyDown}
              className="h-6 text-xs w-32 px-1"
            />
            <Button
              variant="ghost"
              size="sm"
              className="h-6 w-6 p-0"
              onClick={handleSaveTagEdit}
            >
              <Check className="h-3 w-3 text-green-600" />
            </Button>
            <Button
              variant="ghost"
              size="sm"
              className="h-6 w-6 p-0"
              onClick={handleCancelTagEdit}
            >
              <X className="h-3 w-3 text-red-600" />
            </Button>
          </div>
        </div>
      );
    }

    if (isEditingDetails) {
      return (
        <div key={tagId} className="bg-white border rounded p-3 space-y-2 min-w-[200px]">
          <div>
            <label className="text-xs font-medium">Service:</label>
            <div className="text-sm">{tagValue}</div>
          </div>
          
          <div>
            <label className="text-xs font-medium">Famille:</label>
            <Select
              value={editingServiceDetails.family}
              onChange={(e) => setEditingServiceDetails({
                ...editingServiceDetails,
                family: e.target.value
              })}
              className="w-full"
            >
              {serviceFamilies.map(f => (
                <option key={f.id} value={f.id}>{f.name}</option>
              ))}
            </Select>
          </div>
          
          <div>
            <label className="text-xs font-medium">Ordre:</label>
            <Input
              type="number"
              min="1"
              value={editingServiceDetails.order}
              onChange={(e) => setEditingServiceDetails({
                ...editingServiceDetails,
                order: parseInt(e.target.value) || 1
              })}
              className="w-full"
            />
          </div>
          
          <div className="flex gap-1">
            <Button
              variant="ghost"
              size="sm"
              className="h-6 w-6 p-0"
              onClick={handleSaveServiceDetails}
            >
              <Check className="h-3 w-3 text-green-600" />
            </Button>
            <Button
              variant="ghost"
              size="sm"
              className="h-6 w-6 p-0"
              onClick={() => setEditingServiceDetails(null)}
            >
              <X className="h-3 w-3 text-red-600" />
            </Button>
          </div>
        </div>
      );
    }

    return (
      <div key={tagId} className="flex flex-col space-y-1">
        <Badge
          variant="secondary"
          className="px-2 py-1 flex items-center gap-1 cursor-pointer hover:bg-gray-200 group"
          onDoubleClick={() => editingCategory === category && handleStartTagEdit(tag, category)}
        >
          {tagValue}
          {editingCategory === category && (
            <>
              <Button
                variant="ghost"
                size="sm"
                className="h-4 w-4 p-0 opacity-0 group-hover:opacity-100 hover:bg-transparent"
                onClick={(e) => {
                  e.stopPropagation();
                  handleStartTagEdit(tag, category);
                }}
                title="Éditer ce tag"
              >
                <Edit2 className="h-3 w-3 text-blue-600" />
              </Button>
              <Button
                variant="ghost"
                size="sm"
                className="h-4 w-4 p-0 hover:bg-transparent"
                onClick={(e) => {
                  e.stopPropagation();
                  handleRemoveTag(category, tagId);
                }}
                title="Supprimer ce tag"
              >
                <X className="h-3 w-3 text-red-600" />
              </Button>
            </>
          )}
        </Badge>
        
        <div className="text-xs text-gray-500 flex items-center gap-2">
          <span>{familyName} (#{order})</span>
          {editingCategory === category && (
            <Button
              variant="ghost"
              size="sm"
              className="h-4 w-4 p-0 text-blue-600"
              onClick={() => handleEditServiceDetails(tag)}
              title="Modifier famille/ordre"
            >
              <Settings className="h-3 w-3" />
            </Button>
          )}
        </div>
      </div>
    );
  };

  return (
    <div className="p-6 max-w-6xl mx-auto">
      <h2 className="text-2xl font-bold mb-6">Gestion des Tags</h2>

      <Tabs defaultValue="categories" className="w-full">
        <TabsList className="grid w-full grid-cols-2">
          <TabsTrigger value="categories">Tags par catégories</TabsTrigger>
          <TabsTrigger value="services">Organisation des services</TabsTrigger>
        </TabsList>

        <TabsContent value="categories" className="space-y-4">
          {(Object.keys(TAG_CATEGORIES) as TagCategory[]).filter(category => category !== 'priorite').map(category => (
            <Card key={category} className="shadow-sm">
              <CardHeader className="flex flex-row items-center justify-between py-4">
                <CardTitle className="text-lg">{TAG_CATEGORIES[category]}</CardTitle>
                <div className="flex gap-2">
                  {editingCategory === category ? (
                    <>
                      <Button
                        variant="ghost"
                        size="sm"
                        onClick={() => handleSaveEdit(category)}
                      >
                        <Save className="h-4 w-4" />
                      </Button>
                      <Button
                        variant="ghost"
                        size="sm"
                        onClick={() => setEditingCategory(null)}
                      >
                        <X className="h-4 w-4" />
                      </Button>
                    </>
                  ) : (
                    <Button
                      variant="ghost"
                      size="sm"
                      onClick={() => handleStartEdit(category)}
                    >
                      <Edit2 className="h-4 w-4" />
                    </Button>
                  )}
                </div>
              </CardHeader>
              <CardContent>
                {editingCategory === category && (
                  <div className="mb-4">
                    <Button
                      variant="outline"
                      size="sm"
                      onClick={() => setNewTagDialog(true)}
                    >
                      <Plus className="h-4 w-4 mr-2" />
                      Ajouter un tag
                    </Button>
                  </div>
                )}
                <div className="flex flex-wrap gap-2">
                  {Array.isArray(customTags[category]) && customTags[category].map((tag: any) => {
                    if (category === 'services') {
                      return renderServiceTag(tag, category);
                    } else {
                      const tagValue = tag.value || tag;
                      const tagId = tag.id || tag.value || tag;
                      const isCurrentlyEditing = editingTag?.id === tagId;
                      
                      return (
                        <div key={tagId} className="flex items-center">
                          {isCurrentlyEditing ? (
                            <div className="flex items-center gap-1 bg-white border rounded px-2 py-1">
                              <Input
                                ref={editInputRef}
                                value={editingTag.newValue}
                                onChange={(e) => setEditingTag({
                                  ...editingTag,
                                  newValue: e.target.value
                                })}
                                onKeyDown={handleEditKeyDown}
                                className="h-6 text-xs w-32 px-1"
                              />
                              <Button
                                variant="ghost"
                                size="sm"
                                className="h-6 w-6 p-0"
                                onClick={handleSaveTagEdit}
                              >
                                <Check className="h-3 w-3 text-green-600" />
                              </Button>
                              <Button
                                variant="ghost"
                                size="sm"
                                className="h-6 w-6 p-0"
                                onClick={handleCancelTagEdit}
                              >
                                <X className="h-3 w-3 text-red-600" />
                              </Button>
                            </div>
                          ) : (
                            <Badge
                              variant="secondary"
                              className="px-2 py-1 flex items-center gap-1 cursor-pointer hover:bg-gray-200 group"
                              onDoubleClick={() => editingCategory === category && handleStartTagEdit(tag, category)}
                            >
                              {tagValue}
                              {editingCategory === category && (
                                <>
                                  <Button
                                    variant="ghost"
                                    size="sm"
                                    className="h-4 w-4 p-0 opacity-0 group-hover:opacity-100 hover:bg-transparent"
                                    onClick={(e) => {
                                      e.stopPropagation();
                                      handleStartTagEdit(tag, category);
                                    }}
                                    title="Éditer ce tag"
                                  >
                                    <Edit2 className="h-3 w-3 text-blue-600" />
                                  </Button>
                                  <Button
                                    variant="ghost"
                                    size="sm"
                                    className="h-4 w-4 p-0 hover:bg-transparent"
                                    onClick={(e) => {
                                      e.stopPropagation();
                                      handleRemoveTag(category, tagId);
                                    }}
                                    title="Supprimer ce tag"
                                  >
                                    <X className="h-3 w-3 text-red-600" />
                                  </Button>
                                </>
                              )}
                            </Badge>
                          )}
                        </div>
                      );
                    }
                  })}
                </div>
                {editingCategory === category && (
                  <p className="text-xs text-gray-500 mt-2">
                    💡 Double-cliquez sur un tag pour le modifier. {category === 'services' && 'Utilisez l\'onglet "Organisation des services" pour gérer les familles.'}
                  </p>
                )}
              </CardContent>
            </Card>
          ))}
        </TabsContent>

        <TabsContent value="services" className="space-y-4">
          <Card className="shadow-sm">
            <CardHeader>
              <CardTitle className="text-lg">Organisation des services par bulles</CardTitle>
              <p className="text-sm text-gray-600">
                Organisez vos services d'enquête en bulles. Chaque bulle représente un groupe qui apparaîtra sur la page principale.
              </p>
            </CardHeader>
            <CardContent>
              <ServiceFamilyManager
                services={customTags.services || []}
                onUpdateServices={handleUpdateServices}
                onUpdateFamilies={handleUpdateServiceFamilies}
              />
            </CardContent>
          </Card>
        </TabsContent>
      </Tabs>

      <Dialog open={newTagDialog} onOpenChange={setNewTagDialog}>
        <DialogContent>
          <DialogHeader>
            <DialogTitle>
              {editingCategory && `Ajouter un nouveau ${TAG_CATEGORIES[editingCategory].toLowerCase()}`}
            </DialogTitle>
          </DialogHeader>
          
          <div className="space-y-4">
            {editingCategory === 'services' ? (
              <>
                <Input
                  value={newTag}
                  onChange={(e) => setNewTag(e.target.value)}
                  placeholder="Nom du service"
                />
                
                <div>
                  <label className="text-sm font-medium">Famille:</label>
                  <Select
                    value={newServiceFamily}
                    onChange={(e) => setNewServiceFamily(e.target.value)}
                    className="w-full mt-1"
                  >
                    {serviceFamilies.map(f => (
                      <option key={f.id} value={f.id}>{f.name}</option>
                    ))}
                  </Select>
                </div>
                
                <div>
                  <label className="text-sm font-medium">Ordre:</label>
                  <Input
                    type="number"
                    min="1"
                    value={newServiceOrder}
                    onChange={(e) => setNewServiceOrder(parseInt(e.target.value) || 1)}
                    className="w-full mt-1"
                  />
                </div>
              </>
            ) : editingCategory === 'duree' ? (
              <>
                <div>
                  <label className="text-sm font-medium">Type de tag</label>
                  <select
                    className="w-full mt-1 rounded-md border border-gray-300 p-2"
                    value={newTag}
                    onChange={(e) => setNewTag(e.target.value)}
                  >
                    <option value="">Sélectionner un type</option>
                    <option value="enquête à venir">Enquête à venir</option>
                    <option value="enquête de plus de">Enquête de plus de</option>
                  </select>
                </div>
                {newTag === 'enquête de plus de' && (
                  <div>
                    <label className="text-sm font-medium">Durée (en jours)</label>
                    <Input
                      type="number"
                      min="1"
                      value={newDuration}
                      onChange={(e) => setNewDuration(e.target.value)}
                      placeholder="Nombre de jours"
                    />
                  </div>
                )}
              </>
            ) : (
              <Input
                value={newTag}
                onChange={(e) => setNewTag(e.target.value)}
                placeholder={editingCategory ? getPlaceholder(editingCategory) : ''}
              />
            )}
          </div>

          <DialogFooter>
            <Button variant="outline" onClick={() => setNewTagDialog(false)}>
              Annuler
            </Button>
            <Button 
              onClick={() => {
                if (editingCategory === 'duree' && newTag === 'enquête de plus de') {
                  const completeTag = `${newTag} ${newDuration} jours`;
                  handleAddTag(editingCategory, completeTag);
                } else {
                  handleAddTag(editingCategory!);
                }
              }}
            >
              Ajouter
            </Button>
          </DialogFooter>
        </DialogContent>
      </Dialog>
    </div>
  );
};"use client"

import React, { useState, useRef, useEffect } from 'react';
import { Card, CardHeader, CardTitle, CardContent } from '../ui/card';
import { Input } from '../ui/input';
import { Button } from '../ui/button';
import { Badge } from '../ui/badge';
import { Select } from '../ui/select';
import { Edit2, Save, X, Plus, Check, Settings } from 'lucide-react';
import { TAG_CATEGORIES, TagCategory, SERVICE_FAMILIES, ServiceTag, PREDEFINED_TAGS } from '@/config/tags';
import { Dialog, DialogContent, DialogHeader, DialogTitle, DialogFooter } from '../ui/dialog';
import { useAtom } from 'jotai';
import { customTagsAtom } from '@/atoms/tags';
import { useToast } from '@/contexts/ToastContext';
import { ElectronBridge } from '@/utils/electronBridge';
import { APP_CONFIG } from '@/config/constants';

interface TagManagementPageProps {
  onUpdateTags: (category: TagCategory, tags: string[]) => void;
}

interface EditingTag {
  id: string;
  category: TagCategory;
  originalValue: string;
  newValue: string;
}

interface EditingServiceDetails {
  tagId: string;
  family: string;
  order: number;
}

export const TagManagementPage = ({ onUpdateTags }: TagManagementPageProps) => {
  const { showToast } = useToast();
  const [customTags, setCustomTags] = useAtom(customTagsAtom);
  const [editingCategory, setEditingCategory] = useState<TagCategory | null>(null);
  const [editingTag, setEditingTag] = useState<EditingTag | null>(null);
  const [editingServiceDetails, setEditingServiceDetails] = useState<EditingServiceDetails | null>(null);
  const [newTagDialog, setNewTagDialog] = useState(false);
  const [newTag, setNewTag] = useState('');
  const [newDuration, setNewDuration] = useState('');
  const [newServiceFamily, setNewServiceFamily] = useState('sr_pj');
  const [newServiceOrder, setNewServiceOrder] = useState(1);
  const editInputRef = useRef<HTMLInputElement>(null);

  useEffect(() => {
    if (editingTag && editInputRef.current) {
      editInputRef.current.focus();
      editInputRef.current.select();
    }
  }, [editingTag]);

  const getPlaceholder = (category: TagCategory) => {
    switch (category) {
      case 'services':
        return 'Nom du nouveau service d\'enquête';
      case 'infractions':
        return 'Nouveau type d\'infraction';
      case 'duree':
        return 'Nouvelle durée';
      default:
        return 'Nom du tag';
    }
  };

  const handleStartEdit = (category: TagCategory) => {
    setEditingCategory(category);
  };

  const handleSaveEdit = (category: TagCategory) => {
    onUpdateTags(category, customTags[category]);
    setEditingCategory(null);
    showToast('Modifications enregistrées', 'success');
  };

  const handleStartTagEdit = (tag: any, category: TagCategory) => {
    const tagValue = tag.value || tag;
    const tagId = tag.id || tag.value || tag;
    
    setEditingTag({
      id: tagId,
      category,
      originalValue: tagValue,
      newValue: tagValue
    });
  };

  const handleSaveTagEdit = async () => {
    if (!editingTag) return;

    const { id, category, originalValue, newValue } = editingTag;
    
    if (newValue.trim() === '') {
      showToast('Le nom du tag ne peut pas être vide', 'error');
      return;
    }

    if (newValue.trim() === originalValue) {
      setEditingTag(null);
      return;
    }

    const existingTags = customTags[category] || [];
    const tagExists = existingTags.some((tag: any) => {
      const tagValue = tag.value || tag;
      const tagId = tag.id || tag.value || tag;
      return tagValue.toLowerCase() === newValue.trim().toLowerCase() && tagId !== id;
    });

    if (tagExists) {
      showToast('Un tag avec ce nom existe déjà', 'error');
      return;
    }

    try {
      const updatedTags = JSON.parse(JSON.stringify(customTags));
      const tagIndex = updatedTags[category].findIndex((tag: any) => {
        const tagId = tag.id || tag.value || tag;
        return tagId === id;
      });

      if (tagIndex !== -1) {
        if (typeof updatedTags[category][tagIndex] === 'string') {
          updatedTags[category][tagIndex] = {
            id: id,
            value: newValue.trim(),
            category: category
          };
        } else {
          updatedTags[category][tagIndex].value = newValue.trim();
        }
      }

      await updateEnquetesWithNewTagValue(category, originalValue, newValue.trim());
      setCustomTags(updatedTags);
      setEditingTag(null);
      showToast(`Tag "${originalValue}" renommé en "${newValue.trim()}"`, 'success');
    } catch (error) {
      console.error('Erreur lors de la modification du tag:', error);
      showToast('Erreur lors de la modification du tag', 'error');
    }
  };

  const updateEnquetesWithNewTagValue = async (category: TagCategory, oldValue: string, newValue: string) => {
    try {
      const enquetes = await ElectronBridge.getData(APP_CONFIG.STORAGE_KEYS.ENQUETES, []);
      if (!Array.isArray(enquetes)) return;

      let updatedCount = 0;
      const updatedEnquetes = enquetes.map((enquete: any) => {
        if (!enquete.tags || !Array.isArray(enquete.tags)) return enquete;

        const updatedTags = enquete.tags.map((tag: any) => {
          const tagValue = tag.value || tag;
          const tagCategory = tag.category;

          if (tagCategory === category && tagValue === oldValue) {
            updatedCount++;
            return {
              ...tag,
              value: newValue,
              id: tag.id || `${category}-${newValue}`
            };
          }
          return tag;
        });

        return {
          ...enquete,
          tags: updatedTags,
          dateMiseAJour: new Date().toISOString()
        };
      });

      if (updatedCount > 0) {
        await ElectronBridge.setData(APP_CONFIG.STORAGE_KEYS.ENQUETES, updatedEnquetes);
      }
    } catch (error) {
      console.error('Erreur lors de la mise à jour des enquêtes:', error);
      throw error;
    }
  };

  const handleCancelTagEdit = () => {
    setEditingTag(null);
  };

  const handleEditKeyDown = (e: React.KeyboardEvent) => {
    if (e.key === 'Enter') {
      handleSaveTagEdit();
    } else if (e.key === 'Escape') {
      handleCancelTagEdit();
    }
  };

  const handleEditServiceDetails = (tag: any) => {
    const tagValue = tag.value || tag;
    const tagId = tag.id || tag.value || tag;
    
    const serviceTag = PREDEFINED_TAGS.services.find(s => s.value === tagValue);
    const currentFamily = tag.family || serviceTag?.family || 'sr_pj';
    const currentOrder = tag.order || serviceTag?.order || 1;
    
    setEditingServiceDetails({
      tagId: tagId,
      family: currentFamily,
      order: currentOrder
    });
  };

  const handleSaveServiceDetails = async () => {
    if (!editingServiceDetails) return;

    try {
      const updatedTags = JSON.parse(JSON.stringify(customTags));
      const tagIndex = updatedTags.services.findIndex((tag: any) => {
        const tagId = tag.id || tag.value || tag;
        return tagId === editingServiceDetails.tagId;
      });

      if (tagIndex !== -1) {
        if (typeof updatedTags.services[tagIndex] === 'string') {
          updatedTags.services[tagIndex] = {
            id: editingServiceDetails.tagId,
            value: updatedTags.services[tagIndex],
            category: 'services',
            family: editingServiceDetails.family,
            order: editingServiceDetails.order
          };
        } else {
          updatedTags.services[tagIndex] = {
            ...updatedTags.services[tagIndex],
            family: editingServiceDetails.family,
            order: editingServiceDetails.order
          };
        }
      }

      setCustomTags(updatedTags);
      setEditingServiceDetails(null);
      showToast('Détails du service mis à jour', 'success');
    } catch (error) {
      console.error('Erreur lors de la mise à jour:', error);
      showToast('Erreur lors de la mise à jour', 'error');
    }
  };

  const handleAddTag = (category: TagCategory, customTagValue?: string) => {
    const tagToAdd = customTagValue || newTag.trim();
    if (!tagToAdd) return;

    if (category === 'duree') {
      if (!tagToAdd.startsWith('enquête de plus de') && tagToAdd !== 'enquête à venir') {
        showToast('Le tag de durée doit commencer par "enquête de plus de" ou être "enquête à venir"', 'error');
        return;
      }

      if (tagToAdd !== 'enquête à venir' && !tagToAdd.includes('jours') && !newDuration) {
        showToast('Veuillez spécifier une durée', 'error');
        return;
      }
    }

    const existingTags = customTags[category] || [];
    const tagExists = existingTags.some((tag: any) => {
      const tagValue = tag.value || tag;
      return tagValue.toLowerCase() === tagToAdd.toLowerCase();
    });

    if (tagExists) {
      showToast('Ce tag existe déjà', 'error');
      return;
    }

    let newTagObject: any = {
      id: `tag_${Date.now()}_${Math.random().toString(36).substr(2, 9)}`,
      value: tagToAdd,
      category: category
    };

    if (category === 'services') {
      newTagObject.family = newServiceFamily;
      newTagObject.order = newServiceOrder;
    }

    const updatedTagsObject = JSON.parse(JSON.stringify(customTags));
    if (!Array.isArray(updatedTagsObject[category])) {
      updatedTagsObject[category] = [];
    }
    updatedTagsObject[category].push(newTagObject);

    setCustomTags(updatedTagsObject);
    setNewTag('');
    setNewDuration('');
    setNewServiceFamily('sr_pj');
    setNewServiceOrder(1);
    setNewTagDialog(false);
    showToast('Tag ajouté avec succès', 'success');
  };

  const handleRemoveTag = async (category: TagCategory, tagId: string) => {
    try {
      const tagToRemove = customTags[category]?.find((tag: any) => {
        const currentTagId = tag.id || tag.value || tag;
        return currentTagId === tagId;
      });
      
      if (!tagToRemove) {
        showToast('Tag non trouvé', 'error');
        return;
      }

      const tagValue = tagToRemove.value || tagToRemove;
      const enquetes = await ElectronBridge.getData(APP_CONFIG.STORAGE_KEYS.ENQUETES, []);
      const usageCount = enquetes.filter((enquete: any) => 
        enquete.tags?.some((tag: any) => {
          const currentTagValue = tag.value || tag;
          return currentTagValue === tagValue;
        })
      ).length;

      if (usageCount > 0) {
        const confirmDelete = window.confirm(
          `Ce tag est utilisé dans ${usageCount} enquête(s). Êtes-vous sûr de vouloir le supprimer ?`
        );
        if (!confirmDelete) return;

        const updatedEnquetes = enquetes.map((enquete: any) => ({
          ...enquete,
          tags: enquete.tags?.filter((tag: any) => {
            const currentTagValue = tag.value || tag;
            return currentTagValue !== tagValue;
          }) || [],
          dateMiseAJour: new Date().toISOString()
        }));

        await ElectronBridge.setData(APP_CONFIG.STORAGE_KEYS.ENQUETES, updatedEnquetes);
      }

      const updatedTags = JSON.parse(JSON.stringify(customTags));
      if (!updatedTags[category] || !Array.isArray(updatedTags[category])) {
        updatedTags[category] = [];
      } else {
        updatedTags[category] = updatedTags[category].filter((tag: any) => {
          const currentTagId = tag.id || tag.value || tag;
          return currentTagId !== tagId;
        });
      }
      
      setCustomTags(updatedTags);
      showToast(usageCount > 0 ? 
        `Tag supprimé de ${usageCount} enquête(s)` : 
        'Tag supprimé avec succès', 
        'success'
      );
    } catch (error) {
      console.error('Erreur lors de la suppression du tag:', error);
      showToast('Erreur lors de la suppression du tag', 'error');
    }
  };

  const renderServiceTag = (tag: any, category: TagCategory) => {
    const tagValue = tag.value || tag;
    const tagId = tag.id || tag.value || tag;
    const isCurrentlyEditing = editingTag?.id === tagId;
    const isEditingDetails = editingServiceDetails?.tagId === tagId;
    
    const serviceTag = PREDEFINED_TAGS.services.find(s => s.value === tagValue);
    const family = tag.family || serviceTag?.family || 'sr_pj';
    const order = tag.order || serviceTag?.order || 1;
    const familyName = SERVICE_FAMILIES.find(f => f.id === family)?.name || family;

    if (isCurrentlyEditing) {
      return (
        <div key={tagId} className="flex items-center">
          <div className="flex items-center gap-1 bg-white border rounded px-2 py-1">
            <Input
              ref={editInputRef}
              value={editingTag!.newValue}
              onChange={(e) => setEditingTag({
                ...editingTag!,
                newValue: e.target.value
              })}
              onKeyDown={handleEditKeyDown}
              className="h-6 text-xs w-32 px-1"
            />
            <Button
              variant="ghost"
              size="sm"
              className="h-6 w-6 p-0"
              onClick={handleSaveTagEdit}
            >
              <Check className="h-3 w-3 text-green-600" />
            </Button>
            <Button
              variant="ghost"
              size="sm"
              className="h-6 w-6 p-0"
              onClick={handleCancelTagEdit}
            >
              <X className="h-3 w-3 text-red-600" />
            </Button>
          </div>
        </div>
      );
    }

    if (isEditingDetails) {
      return (
        <div key={tagId} className="bg-white border rounded p-3 space-y-2 min-w-[200px]">
          <div>
            <label className="text-xs font-medium">Service:</label>
            <div className="text-sm">{tagValue}</div>
          </div>
          
          <div>
            <label className="text-xs font-medium">Famille:</label>
            <Select
              value={editingServiceDetails.family}
              onChange={(e) => setEditingServiceDetails({
                ...editingServiceDetails,
                family: e.target.value
              })}
              className="w-full"
            >
              {SERVICE_FAMILIES.map(f => (
                <option key={f.id} value={f.id}>{f.name}</option>
              ))}
            </Select>
          </div>
          
          <div>
            <label className="text-xs font-medium">Ordre:</label>
            <Input
              type="number"
              min="1"
              value={editingServiceDetails.order}
              onChange={(e) => setEditingServiceDetails({
                ...editingServiceDetails,
                order: parseInt(e.target.value) || 1
              })}
              className="w-full"
            />
          </div>
          
          <div className="flex gap-1">
            <Button
              variant="ghost"
              size="sm"
              className="h-6 w-6 p-0"
              onClick={handleSaveServiceDetails}
            >
              <Check className="h-3 w-3 text-green-600" />
            </Button>
            <Button
              variant="ghost"
              size="sm"
              className="h-6 w-6 p-0"
              onClick={() => setEditingServiceDetails(null)}
            >
              <X className="h-3 w-3 text-red-600" />
            </Button>
          </div>
        </div>
      );
    }

    return (
      <div key={tagId} className="flex flex-col space-y-1">
        <Badge
          variant="secondary"
          className="px-2 py-1 flex items-center gap-1 cursor-pointer hover:bg-gray-200 group"
          onDoubleClick={() => editingCategory === category && handleStartTagEdit(tag, category)}
        >
          {tagValue}
          {editingCategory === category && (
            <>
              <Button
                variant="ghost"
                size="sm"
                className="h-4 w-4 p-0 opacity-0 group-hover:opacity-100 hover:bg-transparent"
                onClick={(e) => {
                  e.stopPropagation();
                  handleStartTagEdit(tag, category);
                }}
                title="Éditer ce tag"
              >
                <Edit2 className="h-3 w-3 text-blue-600" />
              </Button>
              <Button
                variant="ghost"
                size="sm"
                className="h-4 w-4 p-0 hover:bg-transparent"
                onClick={(e) => {
                  e.stopPropagation();
                  handleRemoveTag(category, tagId);
                }}
                title="Supprimer ce tag"
              >
                <X className="h-3 w-3 text-red-600" />
              </Button>
            </>
          )}
        </Badge>
        
        <div className="text-xs text-gray-500 flex items-center gap-2">
          <span>{familyName} (#{order})</span>
          {editingCategory === category && (
            <Button
              variant="ghost"
              size="sm"
              className="h-4 w-4 p-0 text-blue-600"
              onClick={() => handleEditServiceDetails(tag)}
              title="Modifier famille/ordre"
            >
              <Settings className="h-3 w-3" />
            </Button>
          )}
        </div>
      </div>
    );
  };

  return (
    <div className="p-6 max-w-4xl mx-auto">
      <h2 className="text-2xl font-bold mb-6">Gestion des Tags</h2>

      <div className="space-y-4">
        {(Object.keys(TAG_CATEGORIES) as TagCategory[]).filter(category => category !== 'priorite').map(category => (
          <Card key={category} className="shadow-sm">
            <CardHeader className="flex flex-row items-center justify-between py-4">
              <CardTitle className="text-lg">{TAG_CATEGORIES[category]}</CardTitle>
              <div className="flex gap-2">
                {editingCategory === category ? (
                  <>
                    <Button
                      variant="ghost"
                      size="sm"
                      onClick={() => handleSaveEdit(category)}
                    >
                      <Save className="h-4 w-4" />
                    </Button>
                    <Button
                      variant="ghost"
                      size="sm"
                      onClick={() => setEditingCategory(null)}
                    >
                      <X className="h-4 w-4" />
                    </Button>
                  </>
                ) : (
                  <Button
                    variant="ghost"
                    size="sm"
                    onClick={() => handleStartEdit(category)}
                  >
                    <Edit2 className="h-4 w-4" />
                  </Button>
                )}
              </div>
            </CardHeader>
            <CardContent>
              {editingCategory === category && (
                <div className="mb-4">
                  <Button
                    variant="outline"
                    size="sm"
                    onClick={() => setNewTagDialog(true)}
                  >
                    <Plus className="h-4 w-4 mr-2" />
                    Ajouter un tag
                  </Button>
                </div>
              )}
              <div className="flex flex-wrap gap-2">
                {Array.isArray(customTags[category]) && customTags[category].map((tag: any) => {
                  if (category === 'services') {
                    return renderServiceTag(tag, category);
                  } else {
                    const tagValue = tag.value || tag;
                    const tagId = tag.id || tag.value || tag;
                    const isCurrentlyEditing = editingTag?.id === tagId;
                    
                    return (
                      <div key={tagId} className="flex items-center">
                        {isCurrentlyEditing ? (
                          <div className="flex items-center gap-1 bg-white border rounded px-2 py-1">
                            <Input
                              ref={editInputRef}
                              value={editingTag.newValue}
                              onChange={(e) => setEditingTag({
                                ...editingTag,
                                newValue: e.target.value
                              })}
                              onKeyDown={handleEditKeyDown}
                              className="h-6 text-xs w-32 px-1"
                            />
                            <Button
                              variant="ghost"
                              size="sm"
                              className="h-6 w-6 p-0"
                              onClick={handleSaveTagEdit}
                            >
                              <Check className="h-3 w-3 text-green-600" />
                            </Button>
                            <Button
                              variant="ghost"
                              size="sm"
                              className="h-6 w-6 p-0"
                              onClick={handleCancelTagEdit}
                            >
                              <X className="h-3 w-3 text-red-600" />
                            </Button>
                          </div>
                        ) : (
                          <Badge
                            variant="secondary"
                            className="px-2 py-1 flex items-center gap-1 cursor-pointer hover:bg-gray-200 group"
                            onDoubleClick={() => editingCategory === category && handleStartTagEdit(tag, category)}
                          >
                            {tagValue}
                            {editingCategory === category && (
                              <>
                                <Button
                                  variant="ghost"
                                  size="sm"
                                  className="h-4 w-4 p-0 opacity-0 group-hover:opacity-100 hover:bg-transparent"
                                  onClick={(e) => {
                                    e.stopPropagation();
                                    handleStartTagEdit(tag, category);
                                  }}
                                  title="Éditer ce tag"
                                >
                                  <Edit2 className="h-3 w-3 text-blue-600" />
                                </Button>
                                <Button
                                  variant="ghost"
                                  size="sm"
                                  className="h-4 w-4 p-0 hover:bg-transparent"
                                  onClick={(e) => {
                                    e.stopPropagation();
                                    handleRemoveTag(category, tagId);
                                  }}
                                  title="Supprimer ce tag"
                                >
                                  <X className="h-3 w-3 text-red-600" />
                                </Button>
                              </>
                            )}
                          </Badge>
                        )}
                      </div>
                    );
                  }
                })}
              </div>
              {editingCategory === category && (
                <p className="text-xs text-gray-500 mt-2">
                  💡 Double-cliquez sur un tag pour le modifier. {category === 'services' && 'Utilisez l\'icône ⚙️ pour modifier la famille et l\'ordre.'}
                </p>
              )}
            </CardContent>
          </Card>
        ))}
      </div>

      <Dialog open={newTagDialog} onOpenChange={setNewTagDialog}>
        <DialogContent>
          <DialogHeader>
            <DialogTitle>
              {editingCategory && `Ajouter un nouveau ${TAG_CATEGORIES[editingCategory].toLowerCase()}`}
            </DialogTitle>
          </DialogHeader>
          
          <div className="space-y-4">
            {editingCategory === 'services' ? (
              <>
                <Input
                  value={newTag}
                  onChange={(e) => setNewTag(e.target.value)}
                  placeholder="Nom du service"
                />
                
                <div>
                  <label className="text-sm font-medium">Famille:</label>
                  <Select
                    value={newServiceFamily}
                    onChange={(e) => setNewServiceFamily(e.target.value)}
                    className="w-full mt-1"
                  >
                    {SERVICE_FAMILIES.map(f => (
                      <option key={f.id} value={f.id}>{f.name}</option>
                    ))}
                  </Select>
                </div>
                
                <div>
                  <label className="text-sm font-medium">Ordre:</label>
                  <Input
                    type="number"
                    min="1"
                    value={newServiceOrder}
                    onChange={(e) => setNewServiceOrder(parseInt(e.target.value) || 1)}
                    className="w-full mt-1"
                  />
                </div>
              </>
            ) : editingCategory === 'duree' ? (
              <>
                <div>
                  <label className="text-sm font-medium">Type de tag</label>
                  <select
                    className="w-full mt-1 rounded-md border border-gray-300 p-2"
                    value={newTag}
                    onChange={(e) => setNewTag(e.target.value)}
                  >
                    <option value="">Sélectionner un type</option>
                    <option value="enquête à venir">Enquête à venir</option>
                    <option value="enquête de plus de">Enquête de plus de</option>
                  </select>
                </div>
                {newTag === 'enquête de plus de' && (
                  <div>
                    <label className="text-sm font-medium">Durée (en jours)</label>
                    <Input
                      type="number"
                      min="1"
                      value={newDuration}
                      onChange={(e) => setNewDuration(e.target.value)}
                      placeholder="Nombre de jours"
                    />
                  </div>
                )}
              </>
            ) : (
              <Input
                value={newTag}
                onChange={(e) => setNewTag(e.target.value)}
                placeholder={editingCategory ? getPlaceholder(editingCategory) : ''}
              />
            )}
          </div>

          <DialogFooter>
            <Button variant="outline" onClick={() => setNewTagDialog(false)}>
              Annuler
            </Button>
            <Button 
              onClick={() => {
                if (editingCategory === 'duree' && newTag === 'enquête de plus de') {
                  const completeTag = `${newTag} ${newDuration} jours`;
                  handleAddTag(editingCategory, completeTag);
                } else {
                  handleAddTag(editingCategory!);
                }
              }}
            >
