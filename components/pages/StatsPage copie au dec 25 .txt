import React, { useState, useRef, useEffect } from 'react';
import { useAudience } from '@/hooks/useAudience';
import { Enquete } from '@/types/interfaces';
import { GeneralStats } from '../stats/GeneralStats';
import { AudienceStats } from '../stats/AudienceStats';
import { InfractionStats } from '../stats/InfractionStats';
import { Button } from '../ui/button';
import { AudienceResultModal } from '../modals/AudienceResultModal';
import { useToast } from '@/contexts/ToastContext';
import { ResultatAudience } from '@/types/audienceTypes';
import { ExportPdfButton } from '../pdf/ExportPdfButton';
import { Loader2 } from 'lucide-react';
import { getYearlyStats, getMonthlyStats } from '@/utils/audienceStats';

interface StatsPageProps {
  enquetes: Enquete[];
}

export const StatsPage = ({ enquetes }: StatsPageProps) => {
  const { audienceState, isLoading, saveResultat } = useAudience();
  const { showToast } = useToast();
  const [showDirectResultModal, setShowDirectResultModal] = useState(false);
  const [isExporting, setIsExporting] = useState(false);
  
  // États pour les stats d'audience
  const [selectedYear, setSelectedYear] = useState(new Date().getFullYear());
  const [yearlyStats, setYearlyStats] = useState(null);
  const [monthlyStats, setMonthlyStats] = useState({});
  
  // Référence pour le contenu à exporter
  const contentRef = useRef<HTMLDivElement>(null);
  
  // Calculer les statistiques annuelles et mensuelles
  useEffect(() => {
    if (audienceState?.resultats && enquetes) {
      const stats = getYearlyStats(audienceState.resultats, enquetes, selectedYear);
      setYearlyStats(stats);
      
      // Calculer stats mensuelles
      const currentDate = new Date();
      const lastMonth = selectedYear === currentDate.getFullYear() ? 
        currentDate.getMonth() : 
        11;
      const months = Array.from({ length: lastMonth + 1 }, (_, i) => i);
      const monthlyData = {};
      
      months.forEach(month => {
        monthlyData[month] = getMonthlyStats(audienceState.resultats, enquetes, selectedYear, month);
      });
      
      setMonthlyStats(monthlyData);
    }
  }, [audienceState?.resultats, enquetes, selectedYear]);
  
  if (isLoading) {
    return <div>Chargement des statistiques...</div>;
  }
  
  if (!audienceState?.resultats) {
    return <div>Aucune donnée d'audience disponible</div>;
  }

  const handleSaveDirectResult = async (resultat: ResultatAudience) => {
    try {
      const directResultat = {
        ...resultat,
        isDirectResult: true,
        enqueteId: Date.now() // Génère un ID unique
      };
      
      await saveResultat(directResultat);
      setShowDirectResultModal(false);
      showToast('Résultat enregistré avec succès', 'success');
    } catch (error) {
      showToast('Erreur lors de l\'enregistrement', 'error');
    }
  };

  const allAudienceResults = Object.values(audienceState.resultats);

  return (
    <div className="p-6 max-w-6xl mx-auto space-y-8">
      {/* Bouton d'export en haut à droite avec les props nécessaires */}
      <div className="flex justify-end mb-4">
        <ExportPdfButton 
          contentRef={contentRef} 
          audienceState={audienceState}
          yearlyStats={yearlyStats}
          monthlyStats={monthlyStats}
          selectedYear={selectedYear}
          enquetes={enquetes}
        />
      </div>
      
      {/* Contenu à exporter */}
      <div ref={contentRef}>
        <div className="pdf-section">
          <h3 className="text-xl font-semibold mb-4">Statistiques générales</h3>
          <GeneralStats 
            enquetes={enquetes} 
          />
        </div>

        <div className="pdf-section mt-8">
          <h3 className="text-xl font-semibold mb-4">Types d'infractions</h3>
          <InfractionStats enquetes={enquetes} />
        </div>

        <div className="pdf-section mt-8">
          <h3 className="text-xl font-semibold mb-4">Résultats d'audience</h3>
          <AudienceStats enquetes={enquetes} />
        </div>
      </div>
      
      {/* Bouton pour ajouter une procédure de permanence (hors export PDF) */}
      <div className="flex justify-end mt-4">
        <Button 
          onClick={() => setShowDirectResultModal(true)}
          variant="outline"
        >
          + Ajouter procédure permanence
        </Button>
      </div>

      {showDirectResultModal && (
        <AudienceResultModal
          isOpen={showDirectResultModal}
          onClose={() => setShowDirectResultModal(false)}
          onSave={handleSaveDirectResult}
          enqueteId={Date.now()}
          isDirectResult={true}
        />
      )}
    </div>
  );
};
