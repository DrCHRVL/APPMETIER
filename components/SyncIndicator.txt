// components/SyncIndicator.tsx
import React from 'react';
import { Cloud, CloudOff, Loader2, CheckCircle, AlertTriangle, Users } from 'lucide-react';
import { SyncStatus } from '../types/syncTypes';
import { TooltipProvider, TooltipRoot, TooltipTrigger, TooltipContent } from './ui/tooltip';
import { formatDistanceToNow } from 'date-fns';
import { fr } from 'date-fns/locale';

interface SyncIndicatorProps {
  syncStatus: SyncStatus;
  onClick?: () => void;
  showDetails?: boolean;
}

export const SyncIndicator = ({ 
  syncStatus, 
  onClick, 
  showDetails = true 
}: SyncIndicatorProps) => {
  
  // 🔧 PROTECTION CONTRE SYNCSTATUS UNDEFINED
  if (!syncStatus) {
    const loadingIndicator = {
      icon: <Loader2 className="h-4 w-4 animate-spin" />,
      color: 'text-gray-400',
      bgColor: 'bg-gray-50',
      text: 'Initialisation...'
    };

    if (!showDetails) {
      return (
        <div className={`flex items-center justify-center w-8 h-8 rounded-full transition-colors ${loadingIndicator.bgColor} ${loadingIndicator.color}`}>
          {loadingIndicator.icon}
        </div>
      );
    }

    return (
      <div className={`flex items-center gap-2 px-3 py-1.5 rounded-lg transition-all ${loadingIndicator.bgColor} ${loadingIndicator.color}`}>
        {loadingIndicator.icon}
        <span className="text-sm font-medium">{loadingIndicator.text}</span>
      </div>
    );
  }

  // Déterminer l'icône et la couleur selon le statut
  const getStatusIndicator = () => {
    if (syncStatus.isSync) {
      return {
        icon: <Loader2 className="h-4 w-4 animate-spin" />,
        color: 'text-blue-500',
        bgColor: 'bg-blue-50',
        text: 'Synchronisation...'
      };
    }
    
    if (!syncStatus.isOnline) {
      return {
        icon: <CloudOff className="h-4 w-4" />,
        color: 'text-red-500',
        bgColor: 'bg-red-50',
        text: 'Hors ligne'
      };
    }
    
    if (syncStatus.lastSuccessfulSync) {
      return {
        icon: <CheckCircle className="h-4 w-4" />,
        color: 'text-green-500',
        bgColor: 'bg-green-50',
        text: 'Synchronisé'
      };
    }
    
    return {
      icon: <AlertTriangle className="h-4 w-4" />,
      color: 'text-yellow-500',
      bgColor: 'bg-yellow-50',
      text: 'Attente sync'
    };
  };

  const indicator = getStatusIndicator();
  
  // Formater la dernière synchronisation
  const formatLastSync = () => {
    if (!syncStatus.lastSuccessfulSync) {
      return "Aucune synchronisation";
    }
    
    try {
      const date = new Date(syncStatus.lastSuccessfulSync);
      if (isNaN(date.getTime())) {
        return "Date invalide";
      }
      return formatDistanceToNow(date, { 
        addSuffix: true, 
        locale: fr 
      });
    } catch (error) {
      return "Erreur de date";
    }
  };

  const tooltipContent = (
    <div className="space-y-2">
      <div className="flex items-center gap-2">
        <Users className="h-3 w-3" />
        <span className="text-xs font-medium">{syncStatus.currentUser || 'Utilisateur inconnu'}</span>
      </div>
      
      <div className="text-xs space-y-1">
        <div>
          <span className="font-medium">Statut:</span> {indicator.text}
        </div>
        
        {syncStatus.lastSuccessfulSync && (
          <div>
            <span className="font-medium">Dernière sync:</span> {formatLastSync()}
          </div>
        )}
        
        {syncStatus.lastSyncAttempt && syncStatus.lastSyncAttempt !== syncStatus.lastSuccessfulSync && (
          <div>
            <span className="font-medium">Dernière tentative:</span> {formatDistanceToNow(new Date(syncStatus.lastSyncAttempt), { addSuffix: true, locale: fr })}
          </div>
        )}
        
        {!syncStatus.isOnline && (
          <div className="text-red-600 text-xs mt-1">
            Serveur commun inaccessible
          </div>
        )}
      </div>
      
      {onClick && (
        <div className="text-xs text-gray-500 border-t pt-1 mt-2">
          Cliquer pour synchroniser manuellement
        </div>
      )}
    </div>
  );

  if (!showDetails) {
    // Version compacte - juste l'icône
    return (
      <TooltipProvider>
        <TooltipRoot>
          <TooltipTrigger asChild>
            <button
              onClick={onClick}
              className={`flex items-center justify-center w-8 h-8 rounded-full transition-colors ${indicator.bgColor} ${indicator.color} ${onClick ? 'hover:scale-105 cursor-pointer' : ''}`}
              disabled={syncStatus.isSync}
            >
              {indicator.icon}
            </button>
          </TooltipTrigger>
          <TooltipContent side="bottom">
            {tooltipContent}
          </TooltipContent>
        </TooltipRoot>
      </TooltipProvider>
    );
  }

  // Version détaillée
  return (
    <TooltipProvider>
      <TooltipRoot>
        <TooltipTrigger asChild>
          <button
            onClick={onClick}
            className={`flex items-center gap-2 px-3 py-1.5 rounded-lg transition-all ${indicator.bgColor} ${indicator.color} ${onClick ? 'hover:scale-105 cursor-pointer' : ''} ${syncStatus.isSync ? 'cursor-wait' : ''}`}
            disabled={syncStatus.isSync}
          >
            {indicator.icon}
            <span className="text-sm font-medium">{indicator.text}</span>
            
            {syncStatus.lastSuccessfulSync && !syncStatus.isSync && (
              <span className="text-xs opacity-75">
                ({formatLastSync()})
              </span>
            )}
          </button>
        </TooltipTrigger>
        <TooltipContent side="bottom">
          {tooltipContent}
        </TooltipContent>
      </TooltipRoot>
    </TooltipProvider>
  );
};
